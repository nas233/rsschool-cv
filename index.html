        var chatBox = document.querySelector('.chat-box');
        var chatNavigation = document.querySelector('.chat__navigation');
        var chatRight = document.getElementById('chat_right');

        function getElementHeight(element) {
            return parseInt(window.getComputedStyle(element).height);
        }

        function calculateAvailableHeight() {
            var chatRightRect = chatRight.getBoundingClientRect();
            var windowHeight = window.innerHeight;
            var chatRightHeight = getElementHeight(chatRight);
            var chatNavigationHeight = getElementHeight(chatNavigation);

            return chatRightHeight - chatRightRect.bottom + windowHeight - chatNavigationHeight;
        }

        function updateChatBoxHeight() {
            var availableHeight = calculateAvailableHeight();
            chatBox.style.height = availableHeight + 'px';
        }

        window.addEventListener('scroll', updateChatBoxHeight);
        window.addEventListener('resize', updateChatBoxHeight);
        updateChatBoxHeight();
